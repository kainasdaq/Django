CREATE TABLE keyboard (
	id serial PRIMARY KEY,
	name varchar (50) NOT NULL, 
	type varchar (50) check (type in ('Mechanical', 'Electrostatic Capacitive')),
	color varchar (25) NOT NULL,
	price real NOT NULL,
	CONSTRAINT keyboard_pkey PRIMARY KEY (id)
);
